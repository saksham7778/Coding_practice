class Solution {
public:
    int singleNumber(vector<int>& nums) {
        
        
        int ans =0;int count;
        for(int i=0;i<32;i++){
            count = 0;
            for(int j=0;j<nums.size();j++){
                if(((nums[j]>>i) & 1 ) ==1 ){
                    count++;
                }
            }
            
            if(count%3!=0){
                ans += 1<<i;
            }
        }
        
        return ans;
    }
};
