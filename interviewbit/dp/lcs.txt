int Solution::solve(string a, string b) {
    int n = a.length();int m = b.length();
    
    vector<vector<int>> v(n+1,vector<int>(m+1,0));
    
    for(int i=1;i<=n;i++){
        for(int j=1;j<=m;j++){
            if(a[i-1]==b[j-1]) v[i][j]=v[i-1][j-1]+1;
            else
                v[i][j]=max(v[i-1][j],v[i][j-1]);
        }
    }
    return v[n][m];
}
